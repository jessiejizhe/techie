number to B/M/K

def convert_big_num_smart(num):
    if num >= 1e9:
        big_num = str(round(num / 1e9, 1)) + 'B'
    elif num >= 1e6:
        big_num = str(round(num / 1e6, 1)) + 'M'
    elif num >= 1e3:
        big_num = str(round(num / 1e3, 1)) + 'K'
    else:
        big_num = str(round(num, 1))
    return big_num

def reverseString(s):
    l = s.split(' ')
    out = []
    for ll in l:
        out.append(ll[::-1])
    return ' '.join(out)

def flatten_list(input_list):
    output_list = []
    if len(input_list)>0:
        for i in input_list:
            if isinstance(i, int):
                output_list.append(i)
            else:
                output_list += flatten_list(i)
    return output_list
